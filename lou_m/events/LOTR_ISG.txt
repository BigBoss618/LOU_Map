	major = yes
	
	fire_only_once = yes
	
	#V07
	trigger = {
		always = no
		tag = V49
		is_year = 3018
		is_subject = no

			Z48 = {
				exists = yes
				ai = yes
				was_player = no
				is_subject = no
				is_neighbor_of = ROOT
			}
			V09 = {
				exists = yes

			who = Z48
			modifier = isg_union
		}

			who = V09
			modifier = isg_union
		}

			add_opinion = {
				who = ROOT
				modifier = isg_union

				modifier = isg_union
			}
		}
